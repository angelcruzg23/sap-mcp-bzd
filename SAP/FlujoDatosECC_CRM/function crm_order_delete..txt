function crm_order_delete.
*"----------------------------------------------------------------------
*"*"Lokale Schnittstelle:
*"  IMPORTING
*"     REFERENCE(IT_OBJECTS_TO_DELETE) TYPE  CRMT_OBJECT_GUID_TAB
*"     REFERENCE(IV_UPDATE_TASK_LOCAL) TYPE  BOOLEAN DEFAULT FALSE
*"     REFERENCE(IV_NO_CHECK) TYPE  BOOLEAN DEFAULT FALSE
*"     REFERENCE(IV_NO_BDOC_SEND) TYPE  BOOLEAN DEFAULT FALSE
*"     REFERENCE(IV_CHECK_ONLY) TYPE  BOOLEAN DEFAULT FALSE
*"  EXPORTING
*"     REFERENCE(ET_DELETED_OBJECTS) TYPE  CRMT_RETURN_OBJECTS
*"     REFERENCE(ET_EXCEPTION) TYPE  CRMT_EXCEPTION_T
*"  CHANGING
*"     REFERENCE(CV_LOG_HANDLE) TYPE  BALLOGHNDL OPTIONAL
*"  EXCEPTIONS
*"      SINGLE_DELETION_ERROR
*"      DELETION_ERROR
*"      NO_AUTHORITY
*"      DOCUMENT_NOT_FOUND
*"----------------------------------------------------------------------


  data: lv_collect_exceptions type crmt_boolean.

  if et_exception is requested.
    lv_collect_exceptions = true.
  else.
    lv_collect_exceptions = false.
  endif.

  call function 'CRM_ORDER_DELETE_MULTI_OW'
       exporting
            it_objects_to_delete  = it_objects_to_delete
            iv_update_task_local  = iv_update_task_local
            iv_collect_exceptions = lv_collect_exceptions
            iv_no_check           = iv_no_check
            iv_no_bdoc_send       = iv_no_bdoc_send
            iv_check_only         = iv_check_only
       importing
            et_deleted_objects    = et_deleted_objects
            et_exception          = et_exception
       changing
            cv_log_handle         = cv_log_handle
       exceptions
            single_deletion_error = 1
            no_authority          = 2
            document_not_found    = 3
            others                = 4.

* error handling
  case sy-subrc.

*   no error
    when 0.

*   deletion of one process failed
    when 1.
      message id sy-msgid type sy-msgty number sy-msgno
              with sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4
              raising single_deletion_error.

* no authority
    when 2.
      message id sy-msgid type sy-msgty number sy-msgno
              with sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4
              raising no_authority.

* document not found
    when 3.
      message id sy-msgid type sy-msgty number sy-msgno
              with sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4
              raising document_not_found.

*   other error
    when others.
      message id sy-msgid type sy-msgty number sy-msgno
              with sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4
              raising deletion_error.

  endcase.



endfunction.
